# 逆波兰表达式

对一个不含空格，运算数都是整数，没有负号，且仅涉及 + - * / ( ) 的 C 数学表达式，可以 求出它的计算结果

int eval(const string& expr) {
    stack<int> nums;
    stack<char> ops;
    auto calc = [&nums](char op) {
        int a = nums.top(); nums.pop();
        int b = nums.top(); nums.pop();
        if (op == '+') nums.push(b + a);
        else if (op == '-') nums.push(b - a);
        else if (op == '*') nums.push(b * a);
        else if (op == '/') nums.push(b / a);
    };
    for (int i = 0; i < expr.length(); i++) {
        if (not isdigit(expr[i])) {
            char op = expr[i];
            if (op == '(') ops.push(op);
            else if (op == ')') {
                while (ops.top() != '(') {
                    calc(ops.top()); ops.pop();
                }
                ops.pop();
            }
            else if (op == '+' or op == '-') {
                while (not ops.empty() and (ops.top() == '*' or ops.top() == '/')) {
                    calc(ops.top()); ops.pop();
                }
                ops.push(op);
            }
            else ops.push(op);  // op == '*' or op == '/'
        }
        else {
            int number = 0;
            while (i < expr.length() and isdigit(expr[i])) {
                number = number * 10 + (expr[i] - '0');
                i++;
            }
            i--;
            nums.push(number);
        }
    }
    while (not ops.empty()) {
        calc(ops.top()); ops.pop();
    }
    return nums.top();
}
