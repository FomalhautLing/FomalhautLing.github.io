# 线段树

# 简介

对于一个下标从 1 开始的长为 n 的数组，线段树（Segment Tree）是建立在它之上的一颗完全二叉树，它的每一个叶子节点都是数组中不同下标的一个元素。或者说，线段树是以数组中的元素为叶子，生长起来的一颗二叉树

# 板子

  1. tree：线段树的本体，它是个二叉树，节点编号从 1 开始，它的大小应该为 4n
  2. mark：标记数组，标记打在子树的根上，表示这个子树所有节点都有一个这样的标记
  3. p：线段树节点的编号，若一个节点的编号为 p，则它的左孩子为 2p，右孩子为 2p + 1
  4. cl：该节点维护范围的左端点
  5. cr：该节点维护范围的右端点，包含此端点

这是区间和的板子

struct SegmentTree {
private:
    int n;  // [1, n]
    vector<int> tree;
    vector<int> mark;

    void _build(const vector<int>& base, int l, int r, int p=1) {
        if (l == r) {
            tree[p] = base[l];  // base 从 0 开始的话，这里改为 l - 1
            return;
        }
        int mid = (l + r) / 2;
        _build(base, l, mid, 2*p);
        _build(base, mid + 1, r, 2*p + 1);
        tree[p] = tree[2*p] + tree[2*p + 1];
    }

    void _push_down(int cl, int cr, int p) {
        int mid = (cr + cl) / 2;
        mark[2*p] += mark[p];
        mark[2*p + 1] += mark[p];
        tree[2*p] += mark[p] * (mid - cl + 1);
        tree[2*p + 1] += mark[p] * (cr - mid);
        mark[p] = 0;
    }

    void _update(int l, int r, int k, int cl, int cr, int p=1) {
        if (cr < l or cl > r) return;
        else if (l <= cl and cr <= r) {
            tree[p] += (cr - cl + 1) * k;
            if (cl != cr) mark[p] += k;
        }
        else {
            int mid = (cl + cr) / 2;
            _push_down(cl, cr, p);
            _update(l, r, k, cl, mid, 2*p);
            _update(l, r, k, mid + 1, cr, 2*p + 1);
            tree[p] = tree[2*p] + tree[2*p + 1];
        }
    }

    int _query(int l, int r, int cl, int cr, int p=1) {
        if (cr < l or cl > r) return 0;
        else if (l <= cl and cr <= r) {
            return tree[p];
        }
        else {
            int mid = (cl + cr) / 2;
            _push_down(cl, cr, p);
            return _query(l, r, cl, mid, 2*p) + _query(l, r, mid + 1, cr, 2*p + 1);
        }
    }

public:
    explicit SegmentTree(const vector<int>& base) {
        n = (int) base.size();
        tree.resize(4*n);
        mark.resize(4*n);
        _build(base, 1, n);
    }

    void update(int l, int r, int k) {
        _update(l, r, k, 1, n);
    }

    int query(int l, int r) {
        return _query(l, r, 1, n);
    }
};

私有方法简介

  1. SegmentTree._build(base, l, r, p=1)：递归建树，左节点维护的区间长度是大于等于右节点的
  2. SegmentTree._push_down(cl, cl, p)：下降一层标记，把 p 的标记转给两个孩子，此处的 cl 和 cr 实际上是 p 号节点代表的区间的左右端点（包含），它们必须是准确的
  3. SegmentTree._update(l, r, k, cl, lr, p=1)：给区间 [l, r] 的所有数加上 k，由于 p=1，故只能传入以下取值 cl=1 cr=n，这也是为什么外边还套了一层封装
  4. SegmentTree._query(l, r, cl, lr, p=1)：返回区间 [l, r] 的和，与 _update 一样，依据区间的相交关系（不交，包含，相交）来处理
