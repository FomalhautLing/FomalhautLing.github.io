# 链式前向星

链式前向星是以边的基础的存图方式

struct Graph {
    int n, m, e_cnt{};
    struct Edge {
        int to, next, weight;
        Edge() = default;
        Edge(int to, int next, int weight) : to(to), next(next), weight(weight) {}
    };
    vector<Edge> e;
    vector<int> head;

    Graph(int n, int m) : n(n), m(m) {
        e = vector<Edge>(m + 1, Edge());
        head = vector<int>(m + 1, 0);
    }

    void for_each_child(int from, const function<void(int, int)>& consumer) {
        for (int i = head[from]; i != 0; i = e[i].next) {
            consumer(e[i].to, e[i].weight);
        }
    }

    void add_edge(int from, int to, int weight) {
        e[++e_cnt] = Edge(to, head[from], weight);
        head[from] = e_cnt;
    }

    void clear() {
        for (int i = 1; i <= e_cnt; i++) head[i] = 0;
        e_cnt = 0;
    }
};

# 单源最短路

时间复杂度 ，适用于邻接表，节点编号从 1 开始

注：dijkstra 在有负权的图上无法正常工作

void dijkstra(int from) {
    vector<bool> visited(N, false);
    vector<int> dis(N, inf); dis[from] = 0;
    while (not visited[from]) {
        visited[from] = true;
        for (auto [to, weight] : g[from]) {
            dis[to] = min(dis[to], dis[from] + weight);
        }
        int min_dis = INT_MAX / 2;
        for (int i = 1; i <= n; ++i) {
            if (not visited[i] and dis[i] < min_dis) {
                min_dis = dis[i];
                from = i;
            }
        }
    }
}

优化版，时间复杂度 

vector<int> dijkstra(int from) {
    vector<bool> visited(N, false);
    vector<int> dis(N, inf); dis[from] = 0;

    priority_queue<pair<int, int>, vector<pair<int, int>>, greater<>> pq;
    pq.emplace(0, from);  // pair 的第一个值是 dis[from]

    while (not pq.empty()) {
        from = pq.top().second; pq.pop();
        if (not visited[from]) {
            visited[from] = true;
            for (auto [to, weight] : g[from]) {
                if (dis[to] > dis[from] + weight) {
                    dis[to] = dis[from] + weight;
                    if (not visited[to]) pq.emplace(dis[to], to);
                }
            }
        }
    }
    return dis;
}

# 单源最长路 / 负环

spfa 可以跑单源最长路，把 inf 改为 -inf，同时修改三角不等式即可

vector<int> spfa(int from) {
    vector<bool> in_queue(N, false);
    vector<int> dis(N, inf), visits(N, 0);  // #

    dis[from] = 0;
    queue<int> q; q.push(from);
    in_queue[from] = true;

    while (not q.empty()) {
        from = q.front(); q.pop();
        in_queue[from] = false;
        if (visits[from] > g.n) {
            return vector<int>(0);  // 图有负环
        }
        for (auto [to, weight] : g[from])  {
            if (dis[to] < dis[from] + weight) {  // #
                dis[to] = dis[from] + weight;
                if (not in_queue[to]) {
                    visits[to]++;
                    q.push(to);
                    in_queue[to] = true;
                }
            }
        }
    }
    return dis;
}

# 最小生成树

时间复杂度 ，与 dijkstra 只有一处不同

int prim(int from) {
    int res = 0;
    vector<bool> visited(N, false);
    vector<int> dis(N, inf); dis[from] = 0;
    while (not visited[from]) {
        visited[from] = true;
        for (auto [to, weight] : g[from]) {
            dis[to] = min(dis[to], weight);  // 此处与 dijkstra 不同
        }
        int min_dis = INT_MAX / 2;
        for (int i = 1; i <= n; ++i) {
            if (not visited[i] and dis[i] < min_dis) {
                min_dis = dis[i];
                from = i;
            }
        }
        if (not visited[from]) res += min_dis;
        // while 循环 n 次，建边只要 n - 1 次，最后一次会加上 inf
    }
    return res;
}

# 最小环

时间复杂度 。变量 w 是原图的一份拷贝

基本思路是：floyd 会算出 i 到 j 的最短路（不含 k），如果此时存在边 <j, k> 和 <k, i>，那么这就组成了一个环

int floyd() {
    int res = INT_MAX;
    for (int k = 0; k < n; ++k) {
        for (int i = 0; i < k; ++i) {
            for (int j = 0; j < i; ++j) {
                res = min(res, g[i][j] + w[j][k] + w[k][i]);
            }
        }
        for (int i = 0; i < k; ++i) {
            for (int j = 0; j < k; ++j) {
                g[i][j] = min(g[i][j], g[i][k] + g[k][j]);
            }
        }
    }
    return res;
}

# 强连通分量

强连通分量是对有向图而言的，一个互相可达的点集被成为强连通分量（实际上是构成了环）

  1. dfn：时间戳，记录每个顶点的访问时间，它实际上是 dfs 序
  2. low：最小时间戳，同一个强连通分量的节点 low 相同，就是它们当中最小的 dfn
  3. colour：染色，属于同一个强连通分量的点拥有相同的颜色
  4. cnt：计数器，从 1 开始
  5. col：颜色，从 1 号颜色开始
int dfn[N], low[N], colour[N], cnt, col, stack[N], end_ptr;
void tarjan(int from) {
    dfn[from] = low[from] = ++cnt;
    stack[end_ptr++] = from;        // 一个外部的栈
    for (auto [to, weight] : g) {
        if (dfn[to] == 0) {
            tarjan(to);
            low[from] = min(low[from], low[to]);
        }
        else if (colour[to] == 0) {
            low[from] = min(low[from], low[to]);
        }
    }
    if (dfn[from] == low[from]) {   // from 为强连通分量的根
        colour[from] = ++col;
        while(stack[end_ptr - 1] != from) {  // 染色
            colour[stack[--end_ptr]] = col;
        }
        --end_ptr;
    }
}

# 二分图最大匹配

匈牙利算法的基本思想是，如果后来的顶点没法匹配，那么让前面已匹配的让道

  1. g：邻接矩阵
  2. nl：左侧的顶点数
  3. nr：右侧的顶点数
  4. partner：右侧的顶点 i，与左侧的 partner[i] 匹配

返回该二分图的最大匹配数，左右节点编号从 1 开始

int hungarian(int nl, int nr, bool g[][N]) {
    int n = max(nl, nr);
    vector<bool> visited(n + 1);
    vector<int> partner(n + 1, 0);

    function<bool(int)> matchable = [&](int l) {
        for (int r = 1; r <= nr; ++r)
            if (g[l][r] and not visited[r]) {
                visited[r] = true;
                if (partner[r] == 0 or matchable(partner[r])) {
                    partner[r] = l;
                    return true;
                }
            }
        return false;
    };

    int cnt = 0;
    for (int l = 1; l <= nl; ++l) {
        visited.assign(n, false);
        if (matchable(l)) cnt++;
    }
    return cnt;
}
