# 最小公倍数

最小公倍数（Least Common Multiple. LCM）

性质：lcm(a, b) * gcd(a, b) = a * b

# 扩展欧几里得

求 ax + by = gcd(a, b) 的一组可行解 (x, y)，它的返回值是 gcd(a, b)

int exgcd(int a, int b, int &x, int &y) {
    if (b == 0) {
        x = 1;
        y = 0;
        return a;
    }
    int d = exgcd(b, a % b, y, x);
    y -= (a / b) * x;
    return d;
}

# 扩展欧几里得

若 a 与 P 互质，可以用扩展欧几里得求出逆元

方程 ax = 1 (mod P) 等价于 ax + Py = 1

int inv(int a, int P) {
    int x, y;
    exgcd(a, P, x, y);
    return (x % P + P) % P;
}

# 素数表

素数表一般用欧拉筛生成。欧拉筛是时间复杂度是 的，它会生成 n 以下的素数（不含 n）

vector<int> primes;                     // 素数表
void init(int n) {
    vector<bool> nap(n, false);         // not a prime
    for (int i = 2; i < n; i++) {
        if (not nap[i]) primes.push_back(i);
        for (auto prime : primes) {
            if (prime * i >= n) break;
            nap[prime * i] = true;
            if (i % prime == 0) break;  // 每个数只会被它的最小质因数筛去
        }
    }
}

# 欧拉函数

定义

一个数 n 的欧拉函数可以在 时间内求出

int Phi(int n) {
    int phi = n;
    for (int i = 2; i * i <= n; i++) {
        if (n % i == 0) {
            phi -= phi / i;
            while (n % i == 0) n /= i;
        }
    }
    if (n > 1) phi -= phi / n;
    return phi;
}

# 同余方程组

求解如下形式的方程

结论：记 ，则方程解 

int CRT(int P[], int b[], int n) {
    ll p = 1, x = 0;
    for (int i = 0; i < n; ++i) p *= P[i];
    for (int i = 0; i < n; ++i) {
        ll r = p / P[i];
        x += (b[i] * r * inv(r, P[i])) % p;
    }
    return x % p;
}
